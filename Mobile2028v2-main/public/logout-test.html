How the Fix Works:

const handleLogout = async () => {
    try {
        await logout();
        // Force navigation to home and reload
        window.location.href = '/';
    } catch (error) {
        console.error('Logout error:', error);
        // Even if logout fails, clear local state and redirect
        localStorage.removeItem('auth_token');
        window.location.href = '/';
    }
};
